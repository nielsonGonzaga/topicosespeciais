git --version 		=		Verifica a versão do git;
git init 		= 		Inicia um repositório vazio na pasta atual;
git config --global user.name "seunome"		=	Cria o usuário do git;
git config --global user.email "seuemail"	=	Cria o email do git;
git status 	=	Verifica tudo que não foi salvo;
git add .	=	Coloca os arquivos na área de transferência;
git commit -m "mensagem"	=	Dizer uma mensagem pra identificar o commit;
git log		=	Mostra todos os logs dos commits realizados;
git remote add origin linkDoRepositorio		=	Vincula a pasta local ao repositório do git;
git push -u origin master	=	Fazer commit pela primeira vez.
